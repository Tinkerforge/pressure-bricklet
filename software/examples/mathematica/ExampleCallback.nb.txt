Needs["NETLink`"]
LoadNETAssembly["Tinkerforge",NotebookDirectory[]<>"../../.."]

host="localhost"
port=4223
uid="XYZ"(*Change to your UID*)

(*Create IPConnection and device object*)
ipcon=NETNew["Tinkerforge.IPConnection"]
p=NETNew["Tinkerforge.BrickletPressure",uid,ipcon]
ipcon@Connect[host,port]

(*Callback function for pressure callback (parameter has unit Pa)*)
PressureCB[sender_,pressure_]:=
 Print["Pressure: "<>ToString[N[Quantity[pressure,"kPa/1000"]]]]
AddEventHandler[p@Pressure,PressureCB]

(*Set period for pressure callback to 1s (1000ms)*)
(*Note: The pressure callback is only called every second*)
(*if the pressure has changed since the last call!*)
p@SetPressureCallbackPeriod[1000]

Input["Click OK to exit"]

(*Clean up*)
ipcon@Disconnect[]
ReleaseNETObject[p]
ReleaseNETObject[ipcon]
